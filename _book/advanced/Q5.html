people1.sayName()

混合模式

//混合模式
//mixin
var Person = function(name,age){
  this.name = name
  this.age = age
}
Person.prototype.sayName = function(){
  console.log(this.name)
}

var Student = function(name,age,university){
  Person.call(this,name,age)
  this.university = university
}

Student.prototype = create(Person.prototype)

//重点*
function create(parentObj){
  function F(){}
  F.prototype = parentObj
  return new F()
}

模块模式

//模块模式
//module
var Person = (function(){
  var name = 'tangkalun'
  function sayName(){
    console.log(name)
  }
  return{
    name: name,
    sayName: sayName
  }
})()

工厂模式

//工厂模式
//factory
function createPerson(name){
  var person = {
    name: name,
    sayName: function(){
      console.log(this.name)
    }
  }
  return person
}

createPerson('tangkalun').sayName
createPerson('frankietang').sayName

单例模式

//单例模式（弹窗）
//singleton
//匿名函数 ==> lambda
//这里给大家说一个习惯
//通常看到代码我们先看 return 些什么东西
//instance 是闭包里的变量
var People = (function() {
  var instance
  function init(name) {
    return {
      name: name
    }
  }
  return {
    createPeople: function(name) {
      if(!instance) {
        instance = init(name)
      }
      return instance
    }
  }
}())

People.createPeople('tangkalun')//{name: 'tangkalun'}
People.createPeople('frankietang')//{name: 'tangkalun'}

//发布订阅模式
//publish/subscibe
//jQuery 中的订阅发布模式
$('div').on('click',function(){})
$('header').on('click',function(){
  $('div').trigger('click')

//
var EventCenter = (function(){
  var events = {}
  //('hello',function(){})
  function on(evt,handler){
      /*
      events['hello'] = [{
      handler:function
      }]
      */
    events[evt] = events[evt] || []
    events[evt].push({
      handler: handler
    })
  }
  //('hello')
  function fire(evt,args){
    if(!events[evt]){
      return
    }
    for(var i=0;i<events[evt].length;i++){
      events[evt][i].handler(args)
    }
  }
  function off(name){
    delete events[name ]
  }
  return{
    on: on,
    fire: fire,
    off: off
  }
})()

EventCenter.on('hello',function(){
  console.log('hello')
})
EventCenter.fire('hello')
